Query 1 — Full Delivery Lifecycle Tracking

shipment_lifecycle.sql — Azure Synapse Analytics 
-- Full shipment lifecycle tracking query
-- Tracks each piece through 7+ event types: reception, transfer, hub, last-mile delivery
-- Uses NOT EXISTS subqueries to get only the latest/earliest event per piece
-- Powers the Global Shipment Tracking Dashboard (30+ daily users)

SELECT DISTINCT
    CONVERT(VARCHAR, cp.id_pieza)                      OT,
    CASE
        WHEN CP.cod_embarcador = 103 THEN 'CXP USA'
        WHEN CP.cod_embarcador = 102 THEN 'CXP CHINA'
        WHEN CP.cod_embarcador = 101 THEN 'CXP SPAIN'
    END                                                 Shipper,
    cp.num_guia_aerea                                   MAWB,
    DATEDIFF(DAY, cpd.fecha_pago, TRA_UM.event_date)        DaysPaid_ToLastMile,
    DATEDIFF(DAY, TRA_UM.event_date, ENT.event_date)         DaysLastMile_ToDelivery,
    DATEDIFF(DAY, RECEP_UM4.event_date, ENT.event_date)      DaysInLastMile,
    CASE
        WHEN cd.ind_estado_pago = 1 THEN 'Paid'
        ELSE 'Unpaid'
    END                                                 PaymentStatus,
    CASE
        WHEN SOB.event_description IS NULL THEN 'No Surplus'
        ELSE 'Surplus Found'
    END                                                 SurplusStatus,
    TTD.status_description                              TrackingStatus

FROM idata_cas_pieza CP
LEFT JOIN idata_cas_dips              CD         ON cp.num_id_dips = cd.num_id_dips
LEFT JOIN TransportOrderTrackingEvents UEST       ON cp.id_pieza = UEST.transport_order_number
LEFT JOIN TransportOrderTrackingEvents SOB        ON cp.id_pieza = SOB.transport_order_number
    AND SOB.office_id NOT IN (478, 393, 208, 199, 992) AND SOB.event_code = 16
LEFT JOIN TransportOrderTrackingEvents TRA_UM     ON cp.id_pieza = TRA_UM.transport_order_number
    AND TRA_UM.office_id = 915
LEFT JOIN TransportOrderTrackingEvents RECEP_UM4  ON cp.id_pieza = RECEP_UM4.transport_order_number
    AND RECEP_UM4.office_id NOT IN (478, 393, 208, 199, 992, 915) AND RECEP_UM4.event_code = 4
LEFT JOIN TransportOrderTrackingEvents ENT        ON cp.id_pieza = ENT.transport_order_number
    AND ENT.event_code = 14
LEFT JOIN TransportOrderTrackingData   TTD        ON cp.id_pieza = TTD.transport_order_number
LEFT JOIN idata_cas_pago_dips          CPD        ON cpd.numero_dips = cd.numero_dips

WHERE cp.fec_recepcion >= '20220101'
AND NOT EXISTS (
    -- Ensure we get the latest tracking event per piece only
    SELECT 1 FROM TransportOrderTrackingEvents UEST2
    WHERE cp.id_pieza = UEST2.transport_order_number
    AND UEST2.event_date > UEST.event_date
)
AND NOT EXISTS (
    -- Ensure we get the first surplus event per piece only
    SELECT 1 FROM TransportOrderTrackingEvents SOB2
    WHERE cp.id_pieza = SOB2.transport_order_number
    AND SOB2.event_date > SOB.event_date
    AND SOB2.office_id NOT IN (478, 393, 208, 199, 992) AND SOB2.event_code = 16
)

Query 2 — Hub Reception Validation with Conditional Logic

hub_reception_validation.sql — Azure Synapse Analytics 
-- Hub reception validation query
-- Determines WHERE each order was first received (hub vs field office)
-- Uses nested CASE WHEN + ISNULL to handle multiple reception scenarios
-- Powers the Seller HOT tracking dashboard

SELECT DISTINCT
    TTD.nombre_remitente                                SenderName,
    CONVERT(VARCHAR, SCT.NroOT)                         OT,
    CONVERT(DATE,
        CASE
            WHEN TTD.CREATION_DATE >= TTD.OTD_CREATION_DATE THEN TTD.OTD_CREATION_DATE
            ELSE SCT.FECHACREACION
        END)                                            OrderCreationDate,
    CASE
        WHEN RECEP.OFFICE_ID = 992              THEN 'RECEIVED AT HUB'
        WHEN RECEP.OFFICE_ID IS NULL
            AND FUGA.OFFICE_ID IS NOT NULL      THEN 'RECEIVED OUTSIDE HUB'
        ELSE                                          'PRE-RECEPTION'
    END                                                 HubReceptionStatus,
    CONVERT(DATE,
        ISNULL(RECEP.EVENT_DATE,
            CASE
                WHEN RECEPFATO.EVENT_DATE < FUGA.EVENT_DATE THEN RECEPFATO.EVENT_DATE
                ELSE FUGA.EVENT_DATE
            END))                                    ReceptionDate,
    ISNULL(RECEP.office_name,
        CASE
            WHEN RECEPFATO.EVENT_DATE < FUGA.EVENT_DATE THEN RECEPFATO.office_name
            ELSE FUGA.office_name
        END)                                            ReceptionOffice

FROM      [dbo].[idata_sct_OT]                AS SCT
LEFT JOIN [dbo].[TransportOrderTrackingData]  AS TTD       ON CONVERT(VARCHAR(100), SCT.NroOt) = TTD.transport_order_number
LEFT JOIN [dbo].[TransportOrderTrackingEvents] AS RECEP    ON CONVERT(VARCHAR(100), SCT.NroOt) = RECEP.transport_order_number
    AND RECEP.office_id = 992 AND RECEP.EVENT_CODE = 4 AND RECEP.event_date >= SCT.FECHACREACION
LEFT JOIN [dbo].[TransportOrderTrackingEvents] AS FUGA     ON CONVERT(VARCHAR(100), SCT.NroOt) = FUGA.transport_order_number
    AND FUGA.office_id NOT IN (478, 393, 208, 199) AND FUGA.EVENT_CODE NOT IN (3)

WHERE SCT.NroTCC = 18732331 AND SCT.CodTipoOT = 1
AND (SCT.FECHACREACION >= '2021-01-01' OR TTD.OTD_CREATION_DATE >= '2021-01-01')
AND NOT EXISTS (
    -- Get only the earliest hub reception event per order
    SELECT 1 FROM [dbo].[TransportOrderTrackingEvents] AS RECEP2
    WHERE SCT.NroOT = RECEP2.transport_order_number
    AND RECEP2.event_date >= SCT.FECHACREACION
    AND RECEP2.event_date < RECEP.event_date
    AND RECEP2.office_id = 992 AND RECEP2.EVENT_CODE = 4
)

Query 3 — Billing Records Aggregation

billing_summary.sql — Azure Synapse Analytics 
-- Billing records aggregation query
-- Counts billing events per transport order for a given air waybill
-- Identifies orders with multiple billing records for reconciliation

SELECT
    CP.id_pieza                         OT,
    COUNT(cd.num_id_dips)               BillingRecordsCount

FROM idata_cas_pieza cp
LEFT JOIN idata_cas_dips cd
    ON cp.num_id_dips = cd.num_id_dips

WHERE cp.num_guia_aerea IN (
    '045-46363774',
    '045-46363796'
)

GROUP BY cp.id_pieza
ORDER BY BillingRecordsCount DESC
